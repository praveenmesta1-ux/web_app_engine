import org.cloudbus.cloudsim.*;
import org.cloudbus.cloudsim.core.CloudSim;
import org.cloudbus.cloudsim.provisioners.*;
import java.util.*;

public class FCFS_Simple {

    public static void main(String[] args) {

        try {

            // Step 1: Initialize CloudSim
            CloudSim.init(1, Calendar.getInstance(), false);

            // Step 2: Create Processing Element (CPU)
            List<Pe> peList = new ArrayList<Pe>();
            peList.add(new Pe(0, new PeProvisionerSimple(1000)));

            // Step 3: Create Host
            List<Host> hostList = new ArrayList<Host>();

            Host host = new Host(
                    0,
                    new RamProvisionerSimple(2048),
                    new BwProvisionerSimple(10000),
                    1000000,
                    peList,
                    new VmSchedulerTimeShared(peList)
            );

            hostList.add(host);

            // Step 4: Create Datacenter
            DatacenterCharacteristics dcChar =
                    new DatacenterCharacteristics(
                            "x86", "Linux", "Xen",
                            hostList,
                            10.0, 3.0, 0.05, 0.1, 0.1
                    );

            Datacenter dc = new Datacenter(
                    "Datacenter",
                    dcChar,
                    new VmAllocationPolicySimple(hostList),
                    new LinkedList<Storage>(),
                    0
            );

            // Step 5: Create Broker
            DatacenterBroker broker = new DatacenterBroker("Broker");

            // Step 6: Create VM (SpaceShared = FCFS)
            List<Vm> vmList = new ArrayList<Vm>();

            Vm vm = new Vm(
                    0,
                    broker.getId(),
                    1000,
                    1,
                    512,
                    1000,
                    10000,
                    "Xen",
                    new CloudletSchedulerSpaceShared()
            );

            vmList.add(vm);
            broker.submitVmList(vmList);

            // Step 7: Create Cloudlets (Tasks)
            List<Cloudlet> cloudletList = new ArrayList<Cloudlet>();

            UtilizationModel model = new UtilizationModelFull();

            Cloudlet c1 = new Cloudlet(0, 40000, 1, 300, 300, model, model, model);
            Cloudlet c2 = new Cloudlet(1, 10000, 1, 300, 300, model, model, model);

            c1.setUserId(broker.getId());
            c2.setUserId(broker.getId());

            cloudletList.add(c1);
            cloudletList.add(c2);

            broker.submitCloudletList(cloudletList);

            // Step 8: Start Simulation
            CloudSim.startSimulation();

            List<Cloudlet> result = broker.getCloudletReceivedList();

            CloudSim.stopSimulation();

            // Step 9: Print Results
            for (Cloudlet c : result) {
                System.out.println("Cloudlet " + c.getCloudletId()
                        + " finished at time " + c.getFinishTime());
            }

        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
